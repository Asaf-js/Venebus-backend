    <script>
        // Función para mostrar resultados de búsqueda
        function showResults() {
            document.getElementById('search-results').style.display = 'block';
            document.getElementById('bus-seating').style.display = 'none';
            document.getElementById('passenger-info').style.display = 'none';
            document.getElementById('trip-summary').style.display = 'none';
        }
        
        // Función para mostrar la selección de asientos
        function showBusSeating() {
            document.getElementById('search-results').style.display = 'none';
            document.getElementById('bus-seating').style.display = 'block';
            document.getElementById('passenger-info').style.display = 'none';
            document.getElementById('trip-summary').style.display = 'none';
        }
        
        // Función para seleccionar asientos
        function selectSeat(seat) {
            if (seat.classList.contains('occupied')) return;
            
            seat.classList.toggle('selected');
            seat.classList.toggle('available');
            
            // Actualizar lista de asientos seleccionados
            updateSelectedSeats();
        }
        
        // Actualizar la lista de asientos seleccionados
        function updateSelectedSeats() {
            const selectedSeats = document.querySelectorAll('.seat.selected');
            const seatsText = Array.from(selectedSeats).map(seat => seat.textContent).join(', ');
            document.getElementById('selected-seats').textContent = seatsText || 'Ninguno';
        }
        
        // Función para mostrar información del pasajero
        function showPassengerInfo() {
            document.getElementById('search-results').style.display = 'none';
            document.getElementById('bus-seating').style.display = 'none';
            document.getElementById('passenger-info').style.display = 'block';
            document.getElementById('trip-summary').style.display = 'none';
        }
        
        // Función para mostrar resumen del viaje
        function showTripSummary() {
            // Capturar datos del formulario
            document.getElementById('summary-name').textContent = document.getElementById('name').value;
            document.getElementById('summary-id').textContent = document.getElementById('id-number').value;
            document.getElementById('summary-email').textContent = document.getElementById('email').value;
            document.getElementById('summary-phone').textContent = document.getElementById('phone').value;
            
            document.getElementById('search-results').style.display = 'none';
            document.getElementById('bus-seating').style.display = 'none';
            document.getElementById('passenger-info').style.display = 'none';
            document.getElementById('trip-summary').style.display = 'block';
        }
        
        // Función para confirmar reservación
        function confirmReservation() {
            alert('¡Reserva confirmada! Hemos enviado los detalles a tu correo electrónico.');
        }
        
        // Funciones para cambiar entre vistas
        function showPassengerView() {
            document.querySelector('.passenger-view').style.display = 'block';
            document.getElementById('driver-dashboard').style.display = 'none';
        }
        
        function showDriverView() {
            document.querySelector('.passenger-view').style.display = 'none';
            document.getElementById('driver-dashboard').style.display = 'block';
        }
    </script>
